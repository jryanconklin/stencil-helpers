import colors from '../../../src/tokens/color.json';

# Color Accessibility

The JFOC Brand Colors are loaded via an iframe to this [accessible brand colors](https://abc.useallfive.com) online tool.

<Story name="Accessible Brand Colors">
  {() => {
    const values = Object.values(flattenTokens(colors.color)).join(',').replaceAll('#', '');
    return html`<iframe title="Accessible Brand Colors" width="800" height="800" src="https://abc.useallfive.com/?colors[]=${values}"></iframe>`;
  }}
</Story>
